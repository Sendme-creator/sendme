<script>

  const items = [];

 

  function addItem() {

    const name = document.getElementById('item-name').value.trim();

    const qty = document.getElementById('item-qty').value.trim();

    const desc = document.getElementById('item-desc').value.trim();

 

    if (!name || !qty) {

      alert('Please enter item name and quantity.');

      return;

    }

 

    items.push({ name, qty, desc });

 

    document.getElementById('item-name').value = '';

    document.getElementById('item-qty').value = '';

    document.getElementById('item-desc').value = '';

 

    renderItems();

  }

 

  function removeItem(index) {

    items.splice(index, 1);

    renderItems();

  }

 

  function renderItems() {

    const tbody = document.querySelector('#items-list tbody');

    tbody.innerHTML = '';

 

    items.forEach((item, index) => {

      const tr = document.createElement('tr');

 

      tr.innerHTML = `

        <td>${item.name}</td>

        <td>${item.qty}</td>

        <td>${item.desc || '-'}</td>

        <td><button type="button" class="remove-btn" onclick="removeItem(${index})">X</button></td>

      `;

 

      tbody.appendChild(tr);

    });

  }

 

  function submitOrder(e) {

    e.preventDefault();

    if (items.length === 0) {

      alert('Please add at least one item to your order.');

      return false;

    }

 

    const customerName = document.getElementById('customer-name').value.trim();

    const customerPhone = document.getElementById('customer-phone').value.trim();

    const customerAddress = document.getElementById('customer-address').value.trim();

 

    if (!customerName || !customerPhone || !customerAddress) {

      alert('Please fill in your contact details.');

      return false;

    }

 

    // Prepare order summary

    let orderSummary = `Order from ${customerName}\nPhone: ${customerPhone}\nAddress: ${customerAddress}\n\nItems:\n`;

    items.forEach((item, idx) => {

      orderSummary += `${idx+1}. ${item.name} - Qty: ${item.qty} - ${item.desc || 'No description'}\n`;

    });

 

    // For now, just show an alert with the order details.

    // Later you could send this data to a server or an email service.

    alert('Thank you for your order! Here is the summary:\n\n' + orderSummary);

 

    // Reset form

    items.length = 0;

    renderItems();

    e.target.reset();

 

    return true;

  }

</script>
